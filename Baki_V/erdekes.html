Alapvető SQL Parancsok

CREATE DATABASE

Adatbázis létrehozása

CREATE DATABASE adatbazis_neve;
DROP DATABASE

Adatbázis törlése

DROP DATABASE adatbazis_neve;
CREATE TABLE

Tábla létrehozása

CREATE TABLE hallgatok ( id INT PRIMARY KEY, nev VARCHAR(100), szul_datum DATE );
ALTER TABLE - ADD

Oszlop hozzáadása

ALTER TABLE tabla_neve ADD oszlop_neve tipus;
ALTER TABLE - MODIFY

Oszlop módosítása

ALTER TABLE tabla_neve MODIFY oszlop_neve uj_tipus;
ALTER TABLE - DROP COLUMN

Oszlop törlése

ALTER TABLE tabla_neve DROP COLUMN oszlop_neve;
DROP TABLE

Tábla törlése

DROP TABLE tabla_neve;
SELECT *

Minden oszlop lekérdezése

SELECT * FROM tabla_neve;
